14 def connect(server)
15   opts = server_opts(server)

18   elsif opts.include?(:driver)
19     drv = ::ODBC::Driver.new

21     opts.each do |param, value|
22       if :driver == param && value !~ /\A\{.+\}\z/

24       end

26     end
27     ::ODBC::Database.new.drvconnect(drv)

30   end
